using System;
class Program
{
static void Main()
{
int number = 5;
Console.WriteLine("Factorial of " + number + ": " + Factorial(number));
}
static int Factorial(int n)
{
if (n == 0)
return 1;
else
return n * Factorial(n - 1);
}
}

23. Készíts egy C# programot, amely használ osztályokat és objektumokat egy egyszerű banki tranzakció kezelésére.

class BankTransaction
{
public string AccountHolder { get; set; }
public double Balance { get; set; }
public void Deposit(double amount)
{
Balance += amount;
Console.WriteLine("Deposit of $" + amount + " successful. New balance: $" + Balance);
}
public void Withdraw(double amount)
{
if (amount <= Balance)
{
Balance -= amount;
Console.WriteLine("Withdrawal of $" + amount + " successful. New balance: $" + Balance);
}
else
{
Console.WriteLine("Insufficient funds. Withdrawal failed.");
}
}
}
class Program
{
static void Main()
{
BankTransaction myAccount = new BankTransaction();
myAccount.AccountHolder = "John Doe";
myAccount.Balance = 1000;
Console.WriteLine("Account Holder: " + myAccount.AccountHolder);
Console.WriteLine("Balance: $" + myAccount.Balance);
myAccount.Deposit(500);
myAccount.Withdraw(200);
myAccount.Withdraw(800);
}
}
